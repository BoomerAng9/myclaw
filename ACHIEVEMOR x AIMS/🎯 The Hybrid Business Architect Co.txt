## 🔐 Authentication Implementation (2026 Best Practices)

### Authentication Architecture[web:249][web:252][web:255]

**Chosen Solution: Better Auth + Multi-Provider OAuth**

// lib/auth.config.ts
import { betterAuth } from 'better-auth';
import { firebaseAdapter } from 'better-auth/adapters/firebase';
import { passkey } from 'better-auth/plugins/passkey';

export const auth = betterAuth({
  database: firebaseAdapter({
    credential: process.env.GOOGLE_APPLICATION_CREDENTIALS,
    projectId: process.env.FIREBASE_PROJECT_ID,
  }),
  
  // Multi-provider configuration
  socialProviders: {
    google: {
      clientId: process.env.GOOGLE_CLIENT_ID!,
      clientSecret: process.env.GOOGLE_CLIENT_SECRET!,
    },
    telegram: {
      botToken: process.env.TELEGRAM_BOT_TOKEN!,
    },
    whatsapp: {
      phoneNumberId: process.env.WHATSAPP_PHONE_NUMBER_ID!,
      accessToken: process.env.WHATSAPP_ACCESS_TOKEN!,
    },
  },
  
  // Phishing-resistant passkeys[web:252]
  plugins: [
    passkey({
      rpName: 'Hybrid Business Architect',
      rpID: process.env.NEXT_PUBLIC_APP_DOMAIN || 'localhost',
    }),
  ],
  
  // Session configuration for App Router[web:249]
  session: {
    cookieCache: {
      enabled: true,
      maxAge: 60 * 5, // 5 minutes
    },
    expiresIn: 60 * 60 * 24 * 7, // 7 days
  },
  
  // Edge-compatible session validation
  advanced: {
    useSecureCookies: process.env.NODE_ENV === 'production',
    cookiePrefix: 'hba_auth',
  },
});

export const { signIn, signOut, session } = auth;

**Why Better Auth over NextAuth.js (2026):**[web:249][web:252]
- Native App Router support with Server Components
- Edge runtime compatible
- Built-in passkey (WebAuthn) support
- Simpler session validation in middleware
- Better TypeScript types

### Middleware for Route Protection[web:255][web:258]

// middleware.ts
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';
import { auth } from './lib/auth.config';

export async function middleware(request: NextRequest) {
  const { pathname } = request.nextUrl;
  
  // Public routes
  const isPublicRoute = pathname === '/' || pathname === '/auth';
  
  // Get session (validates cookie server-side)[web:252]
  const session = await auth.session({ request });
  
  // Redirect unauthenticated users to auth
  if (!isPublicRoute && !session) {
    return NextResponse.redirect(new URL('/auth', request.url));
  }
  
  // Redirect authenticated users away from auth page
  if (isPublicRoute && session) {
    return NextResponse.redirect(new URL('/hub', request.url));
  }
  
  return NextResponse.next();
}

function AnimatedParticles({ count }: { count: number }) {
  return (

      {Array.from({ length: count }).map((_, i) => (
        <motion.div
          key={i}
          className="absolute w-1 h-1 bg-gold-bright/30 rounded-full"
          initial={{
            x: Math.random() * window.innerWidth,
            y: Math.random() * window.innerHeight
          }}
          animate={{
            y: [null, -20, 0],
            opacity: [0.3, 0.6, 0.3]
          }}
          transition={{
            duration: 3 + Math.random() * 2,
            repeat: Infinity,
            delay: Math.random() * 2
          }}
        />
      ))}

  );
}

// backend/src/routes/auth.ts
import { Router } from 'express';
import { firebaseAdmin } from '../services/firebase';

const router = Router();

router.post('/login', async (req, res) => {
  const { provider, credential } = req.body;
  
  try {
    // Verify Firebase ID token
    const decodedToken = await firebaseAdmin.auth().verifyIdToken(credential);
    
    // Create/update user document
    const userRef = firebaseAdmin.firestore().collection('users').doc(decodedToken.uid);
    
    await userRef.set({
      uid: decodedToken.uid,
      provider,
      email: decodedToken.email,
      createdAt: new Date(),
      lastLogin: new Date(),
      onboardingComplete: false,
      profile: null
    }, { merge: true });
    
    // Generate session token
    const sessionToken = await firebaseAdmin.auth().createCustomToken(decodedToken.uid);
    
    res.json({
      success: true,
      userId: decodedToken.uid,
      sessionToken
    });
  } catch (error) {
    console.error('Authentication error:', error);
    res.status(401).json({ 
      error: 'Authentication failed',
      message: error.message 
    });
  }
});

export default router;

### 2. Configure Tailwind CSS

Create `frontend/tailwind.config.js`:

/** @type {import('tailwindcss').Config} */
module.exports = {
  content: [
    './app/**/*.{js,ts,jsx,tsx,mdx}',
    './components/**/*.{js,ts,jsx,tsx,mdx}',
  ],
  theme: {
    extend: {
      colors: {
        obsidian: {
          DEFAULT: '#1f2121',
          light: '#2a2c2c',
          dark: '#0d0d0d',
        },

          emboss: '#d4af37',
          bright: '#f4d03f',
          dim: '#8b7329',
        },
      },

        doto: ['Doto', 'monospace'],
        sans: ['Inter', 'system-ui', 'sans-serif'],
      },
      backdropBlur: {
        glass: '20px',
      },
      boxShadow: {
        glass: '0 8px 32px rgba(0, 0, 0, 0.37)',
        glow: '0 0 20px rgba(244, 208, 63, 0.5)',
      },
    },
  },
  plugins: [],
}

### 4. Create Global CSS

Create `frontend/app/globals.css`:

@tailwind base;
@tailwind components;
@tailwind utilities;

@font-face {
  font-family: 'Doto';
  src: url('/fonts/Doto.woff2') format('woff2');
  font-weight: 400 700;
  font-display: swap;
  font-style: normal;
}

@layer base {
  * {
    @apply border-border;
  }
  body {
    @apply bg-obsidian text-white;
  }
}

@layer components {
  .glass-panel {
    @apply bg-white/5 backdrop-blur-glass rounded-2xl border border-gold-emboss/20 shadow-glass;
  }
}

### 5. Create Root Layout

Create `frontend/app/layout.tsx`:

import type { Metadata } from 'next'
import { Inter } from 'next/font/google'
import './globals.css'

const inter = Inter({ 
  subsets: ['latin'],
  variable: '--font-inter'
})

export const metadata: Metadata = {
  title: 'Hybrid Business Architect | ACHEEVY',
  description: 'AI-powered business strategy platform',
}

export default function RootLayout({
  children,
}: {
  children: React.ReactNode
}) {
  return (
    <html lang="en" className={inter.variable}>
      <head>
        <link
          href="https://fonts.googleapis.com/css2?family=Doto&display=swap"
          rel="stylesheet"
        />
      </head>
      <body className="font-sans antialiased">{children}</body>
    </html>
  )
}

### Firebase Security Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    match /tasks/{taskId} {
      allow read: if request.auth != null && 
                     resource.data.userId == request.auth.uid;
      allow write: if request.auth != null;
    }
  }
}

## 🆘 Troubleshooting Guide

### Common Issues

**1. Tailwind classes not applying:**
# Ensure content paths include all component directories
# Check tailwind.config.js content array
# Restart dev server after config changes

**2. Font not loading:**
# Verify font file exists in /public/fonts/
# Check @font-face declaration in globals.css
# Clear browser cache

**3. Glassmorphic effect not working:**
# Ensure backdrop-filter is supported (requires HTTPS in production)
# Check browser compatibility
# Verify backdrop-blur utility is configured in Tailwind

**4. Firebase authentication errors:**
# Verify API keys in .env.local
# Check Firebase console for enabled providers
# Ensure authorized domains include localhost
